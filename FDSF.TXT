    <script>
        const STORAGE_KEY = 'nss_attendance_final';
        const rollInput = document.getElementById('rollInput');
        const saveBtn = document.getElementById('saveBtn');
        const exportBtn = document.getElementById('exportBtn');
        const clearBtn = document.getElementById('clearBtn');
        const scanBtn = document.getElementById('scanBtn');
        const stopBtn = document.getElementById('stopBtn');
        const alertBox = document.getElementById('alertBox');
        const scannerContainer = document.getElementById('scanner-container');
        const tableBody = document.getElementById('attendanceTableBody');
        const countLabel = document.getElementById('count');

        let html5QrCode = null;
        let todayAttendance = new Set();
        let isScanning = false;

        function notify(msg, type = 'success') {
            alertBox.textContent = msg;
            alertBox.className = `alert ${type}`;
            alertBox.style.display = 'block';
            setTimeout(() => { alertBox.style.display = 'none'; }, 4000);
        }

        // --- ROW CREATION WITH 3-DOTS MENU ---
        function addRowToTable(roll, timestamp) {
            const row = document.createElement('tr');
            const timeStr = new Date(timestamp).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
            
            // Unique ID for the dropdown
            const menuId = 'menu-' + roll.replace(/[^a-zA-Z0-9]/g, '');

            row.innerHTML = `
                <td class="roll-col">${roll}</td>
                <td class="time-col">
                    ${timeStr}
                    <div class="menu-container">
                        <button class="menu-btn" onclick="toggleMenu('${menuId}')">â‹®</button>
                        <div id="${menuId}" class="dropdown">
                            <button class="delete-option" onclick="deleteEntry('${roll}', this)">Delete</button>
                        </div>
                    </div>
                </td>
            `;
            tableBody.insertBefore(row, tableBody.firstChild);
            countLabel.textContent = todayAttendance.size;
        }

        // --- TOGGLE MENU ---
        window.toggleMenu = function(menuId) {
            // Close all other menus first
            document.querySelectorAll('.dropdown').forEach(d => {
                if (d.id !== menuId) d.classList.remove('show');
            });
            const menu = document.getElementById(menuId);
            if (menu) menu.classList.toggle('show');
        }

        // Close menu when clicking outside
        window.onclick = function(event) {
            if (!event.target.matches('.menu-btn')) {
                document.querySelectorAll('.dropdown').forEach(d => d.classList.remove('show'));
            }
        }

        // --- DELETE SINGLE ENTRY ---
        window.deleteEntry = function(roll, btnElement) {
            if(confirm(`Delete ${roll} from list?`)) {
                // 1. Remove from LocalStorage
                let data = getStoredData();
                const today = new Date().toISOString().split('T')[0];
                
                // Filter out the specific item (Roll AND Today's date)
                const newData = data.filter(record => !(record.roll === roll && record.date === today));
                localStorage.setItem(STORAGE_KEY, JSON.stringify(newData));

                // 2. Remove from Set
                todayAttendance.delete(roll);

                // 3. Remove row from UI
                // We find the closest 'tr' (table row) and remove it
                const row = btnElement.closest('tr');
                row.remove();

                // 4. Update count
                countLabel.textContent = todayAttendance.size;
                notify(`Deleted ${roll}`, "warning");
            }
        }

        // --- CLEAR ALL DATA ---
        clearBtn.addEventListener('click', () => {
            if(confirm("ARE YOU SURE?\n\nThis will delete all scanned numbers from this phone permanently.")) {
                localStorage.removeItem(STORAGE_KEY);
                todayAttendance.clear();
                tableBody.innerHTML = '';
                countLabel.textContent = '0';
                notify("All data cleared.", "warning");
            }
        });

        // --- Scanner Logic ---
        scanBtn.addEventListener('click', async () => {
            if (location.protocol !== 'https:' && location.hostname !== 'localhost') {
                notify("Camera blocked! Use HTTPS or GitHub Pages.", "error");
                return;
            }
            if (typeof Html5Qrcode === "undefined") {
                notify("Library loading... check internet.", "warning");
                return;
            }

            try {
                if (!html5QrCode) html5QrCode = new Html5Qrcode("reader");
                scannerContainer.classList.add('active');
                scanBtn.style.display = 'none';
                stopBtn.style.display = 'block';
                
                const config = { 
                    fps: 10, 
                    qrbox: { width: 250, height: 250 },
                    aspectRatio: 1.0,
                    formatsToSupport: [ 
                        Html5QrcodeSupportedFormats.QR_CODE,
                        Html5QrcodeSupportedFormats.CODE_128,
                        Html5QrcodeSupportedFormats.CODE_39,
                        Html5QrcodeSupportedFormats.EAN_13
                    ]
                };

                await html5QrCode.start({ facingMode: "environment" }, config, onScanSuccess, onScanError);
                isScanning = true;

            } catch (err) {
                notify("Camera error: " + err, "error");
                stopScanner();
            }
        });

        function onScanSuccess(decodedText) {
            handleRollNumber(decodedText, true);
        }
        function onScanError(error) {}

        stopBtn.addEventListener('click', stopScanner);

        function stopScanner() {
            if (html5QrCode && isScanning) {
                html5QrCode.stop().then(() => {
                    scannerContainer.classList.remove('active');
                    scanBtn.style.display = 'block';
                    stopBtn.style.display = 'none';
                    isScanning = false;
                    html5QrCode.clear();
                });
            } else {
                 scannerContainer.classList.remove('active');
                 scanBtn.style.display = 'block';
                 stopBtn.style.display = 'none';
            }
        }

        // --- Data Handling ---
        function getStoredData() {
            const data = localStorage.getItem(STORAGE_KEY);
            return data ? JSON.parse(data) : [];
        }

        function handleRollNumber(text, isScanned = false) {
            const cleanText = text.trim().toUpperCase();
            
            if (!cleanText) {
                if(!isScanned) notify("Enter a value", "error"); 
                return; 
            }

            if (!isScanned) rollInput.value = cleanText;

            if (todayAttendance.has(cleanText)) {
                notify(cleanText + " already scanned!", "warning");
                if(!isScanned) rollInput.value = '';
                return;
            }

            const timestamp = Date.now();
            const record = { 
                roll: cleanText, 
                date: new Date().toISOString().split('T')[0], 
                timestamp: timestamp 
            };
            
            const data = getStoredData();
            data.push(record);
            localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
            
            todayAttendance.add(cleanText);
            addRowToTable(cleanText, timestamp);
            notify("Saved: " + cleanText, "success");
            
            if (!isScanned) {
                rollInput.value = '';
                rollInput.focus();
            } else {
                rollInput.value = cleanText;
            }
        }

        saveBtn.addEventListener('click', () => handleRollNumber(rollInput.value));
        
        rollInput.addEventListener('keydown', (e) => {
            if (e.key === 'Enter') handleRollNumber(rollInput.value);
        });

        exportBtn.addEventListener('click', () => {
            const data = getStoredData();
            if (data.length === 0) {
                notify("No data to export", "warning");
                return;
            }
            const csvContent = "ID,Date,Time\n" + 
                data.map(r => `${r.roll},${r.date},${new Date(r.timestamp).toLocaleTimeString()}`).join("\n");

            const blob = new Blob([csvContent], { type: 'text/csv' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `attendance_${new Date().toISOString().split('T')[0]}.csv`;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
        });

        // USB Scanner Support
        let kbBuffer = '';
        let kbTimer;
        document.addEventListener('keydown', (e) => {
            if (e.target.tagName === 'INPUT') return;
            if (e.key === 'Enter') {
                if (kbBuffer.length > 2) handleRollNumber(kbBuffer, true);
                kbBuffer = '';
            } else {
                if (e.key.length === 1) kbBuffer += e.key;
                clearTimeout(kbTimer);
                kbTimer = setTimeout(() => kbBuffer = '', 200);
            }
        });

        // Init
        function loadTodayData() {
            const today = new Date().toISOString().split('T')[0];
            const data = getStoredData();
            todayAttendance.clear();
            tableBody.innerHTML = ''; 
            const todaysRecords = data.filter(record => record.date === today);
            todaysRecords.forEach(record => {
                todayAttendance.add(record.roll);
                addRowToTable(record.roll, record.timestamp);
            });
            countLabel.textContent = todayAttendance.size;
        }
        loadTodayData();
        
        if (location.protocol !== 'https:' && location.hostname !== 'localhost') {
            notify("Camera requires HTTPS!", "error");
        }
    </script>
